#include<iostream>
using namespace std;

void insertasc(char* arr, char a) {
	int i = strlen(arr)-1;
	while (i >= 0) {
		if (a < arr[i]) {
			arr[i + 1] = arr[i];
			arr[i] = a;
			--i;
		}
		else
			break;
	}
	printf("ÉıĞò£º%s\n", arr);
}

			arr[i + 1] = arr[i];
			arr[i] = a;
			--i;
		}

int main() {
	char a = 0;

	char arr2[256] = "864321";
