#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
int main()
{
	int num;
	printf("ÇëÊäÈë×Ö·û£º\n");
	scanf("%d", &num);
	while ((num = getchar()) != EOF)
	{	
		if (num >= 'a'&&num <= 'z')
		{
			num = num - 32;
			printf("%c\n", num);	}

		else if (num >= 'A'&&num <= 'Z')
		{
			num = num + 32;
			printf("%c\n",num );
		}

	}
	system("pause");
	return 0;
}
